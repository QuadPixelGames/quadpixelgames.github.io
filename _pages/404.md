<script>
(function() {
  const canvas = document.getElementById('gameCanvas');
  const ctx = canvas.getContext('2d');
  const scoreDisplay = document.getElementById('scoreDisplay');
  
  let player = { x: 300, y: 350, size: 20 };
  let enemies = [];
  let score = 0;
  let gameLoop;
  let isGameOver = false;
  let frameCount = 0;

  // Track mouse position relative to canvas
  canvas.addEventListener('mousemove', function(e) {
    const rect = canvas.getBoundingClientRect();
    const scaleX = canvas.width / rect.width;
    const scaleY = canvas.height / rect.height;
    player.x = (e.clientX - rect.left) * scaleX - player.size / 2;
    player.y = (e.clientY - rect.top) * scaleY - player.size / 2;
    
    // Keep player in bounds
    if (player.x < 0) player.x = 0;
    if (player.x + player.size > canvas.width) player.x = canvas.width - player.size;
    if (player.y < 0) player.y = 0;
    if (player.y + player.size > canvas.height) player.y = canvas.height - player.size;
  });

  function spawnEnemy() {
    const size = Math.random() * 30 + 15;
    enemies.push({
      x: Math.random() * (canvas.width - size),
      y: -size,
      size: size,
      speed: Math.random() * 3 + 2 + (score / 500) // Increase speed as score goes up
    });
  }

  function resetGame() {
    enemies = [];
    score = 0;
    frameCount = 0;
    isGameOver = false;
    update();
  }

  function update() {
    if (isGameOver) return;
    
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // Draw Player
    ctx.fillStyle = '#00ff88';
    ctx.fillRect(player.x, player.y, player.size, player.size);
    ctx.shadowBlur = 10;
    ctx.shadowColor = '#00ff88';
    
    // Spawn enemies
    frameCount++;
    if (frameCount % Math.max(20, 60 - Math.floor(score / 100)) === 0) {
      spawnEnemy();
    }

    // Update and Draw Enemies
    ctx.shadowBlur = 0;
    ctx.fillStyle = '#ff4444';
    for (let i = 0; i < enemies.length; i++) {
      let e = enemies[i];
      e.y += e.speed;
      ctx.fillRect(e.x, e.y, e.size, e.size);
      
      // Collision detection
      if (player.x < e.x + e.size &&
          player.x + player.size > e.x &&
          player.y < e.y + e.size &&
          player.y + player.size > e.y) {
        gameOver();
      }
    }
    
    // Remove off-screen enemies
    enemies = enemies.filter(e => e.y < canvas.height);
    
    score++;
    if (score % 10 === 0) {
      scoreDisplay.innerText = 'Score: ' + Math.floor(score / 10);
    }
    
    gameLoop = requestAnimationFrame(update);
  }

  function gameOver() {
    isGameOver = true;
    cancelAnimationFrame(gameLoop);
    ctx.fillStyle = 'rgba(0,0,0,0.7)';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = '#ff4444';
    ctx.font = 'bold 30px "Outfit", sans-serif';
    ctx.textAlign = 'center';
    ctx.fillText('GLITCHED OUT!', canvas.width / 2, canvas.height / 2 - 20);
    
    ctx.fillStyle = '#fff';
    ctx.font = '20px "Outfit", sans-serif';
    ctx.fillText('Click to restart', canvas.width / 2, canvas.height / 2 + 20);
  }

  canvas.addEventListener('click', function() {
    if (isGameOver) resetGame();
  });

  // Start
  update();
})();
</script>
